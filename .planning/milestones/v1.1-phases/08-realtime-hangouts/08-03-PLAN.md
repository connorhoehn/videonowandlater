```typescript
// From IVS Recording State Change events
interface RecordingEndEvent {
  detail: {
    resource_arn: string;  // Channel ARN or Stage ARN
    recording_s3_key_prefix: string;
    recording_duration_ms: number;
    recording_status: 'Recording End' | 'Recording End Failure';
  };
}

// ARN formats:
// Channel: arn:aws:ivs:us-east-1:123456789012:channel/abcdefg
// Stage:   arn:aws:ivs:us-east-1:123456789012:stage/xyzabc
```
