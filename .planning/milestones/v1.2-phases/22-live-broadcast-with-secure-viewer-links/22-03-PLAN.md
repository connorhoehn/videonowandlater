```typescript
import type { APIGatewayProxyHandler } from 'aws-lambda';
import { ScanCommand } from '@aws-sdk/lib-dynamodb';
import { getDocumentClient } from '../lib/dynamodb-client';
import { SessionStatus } from '../domain/session';

export const handler: APIGatewayProxyHandler = async (event) => {
  const tableName = process.env.TABLE_NAME!;
  const userId = event.requestContext?.authorizer?.claims?.['cognito:username'];

  const docClient = getDocumentClient();

  // Scan all ended sessions
  const scanResult = await docClient.send(
    new ScanCommand({
      TableName: tableName,
      FilterExpression: '#status = :ended',
      ExpressionAttributeNames: {
        '#status': 'status',
      },
      ExpressionAttributeValues: {
        ':ended': SessionStatus.ENDED,
      },
    })
  );

  let sessions = scanResult.Items || [];

  // Filter private sessions: only show to owner
  if (userId) {
    sessions = sessions.filter((session: any) => {
      // Public sessions (isPrivate is false or undefined) are visible to everyone
      if (!session.isPrivate) {
        return true;
      }
      // Private sessions are only visible to the owner
      return session.userId === userId;
    });
  } else {
    // No auth: only show public sessions
    sessions = sessions.filter((session: any) => !session.isPrivate);
  }

  // Sort by createdAt descending (most recent first)
  sessions.sort((a: any, b: any) => {
    return new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime();
  });

  return {
    statusCode: 200,
    body: JSON.stringify(sessions),
  };
};
```

Key changes:
- Extract userId from Cognito token (authorizer.claims['cognito:username'])
- Add filter that checks isPrivate and userId equality
- Preserve existing sorting logic
- Treat undefined isPrivate as false (backward compatible)
- If no auth token, only show public sessions
