Create useBroadcast.ts hook following research Pattern 1:
```typescript
import { useState, useEffect, useRef } from 'react';
import IVSBroadcastClient, { BASIC_FULL_HD_LANDSCAPE } from 'amazon-ivs-web-broadcast';

export function useBroadcast(sessionId: string, apiBaseUrl: string, authToken: string) {
  const [client, setClient] = useState<IVSBroadcastClient | null>(null);
  const [isLive, setIsLive] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const previewRef = useRef<HTMLVideoElement>(null);

  // Initialize client on mount
  useEffect(() => {
    const initClient = async () => {
      try {
        const response = await fetch(`${apiBaseUrl}/sessions/${sessionId}/start`, {
          method: 'POST',
          headers: { Authorization: `Bearer ${authToken}` },
        });
        const { ingestEndpoint } = await response.json();

        const broadcastClient = IVSBroadcastClient.create({
          streamConfig: BASIC_FULL_HD_LANDSCAPE,
          ingestEndpoint,
        });
        setClient(broadcastClient);

        if (previewRef.current) {
          broadcastClient.attachPreview(previewRef.current);
        }
      } catch (err: any) {
        setError(err.message);
      }
    };
    initClient();

    return () => {
      if (client) client.stopBroadcast();
    };
  }, [sessionId, apiBaseUrl, authToken]);

  const startBroadcast = async () => {
    if (!client) return;
    try {
      // Get camera/mic
      const stream = await navigator.mediaDevices.getUserMedia({
        video: { width: { ideal: 1920 }, height: { ideal: 1080 } },
        audio: true,
      });

      client.addVideoInputDevice(stream, 'camera1', { index: 0 });
      const audioTrack = stream.getAudioTracks()[0];
      if (audioTrack) {
        const micStream = new MediaStream([audioTrack]);
        client.addAudioInputDevice(micStream, 'mic1');
      }

      // Fetch stream key
      const response = await fetch(`${apiBaseUrl}/sessions/${sessionId}/start`, {
        method: 'POST',
        headers: { Authorization: `Bearer ${authToken}` },
      });
      const { streamKey } = await response.json();

      await client.startBroadcast(streamKey);
      setIsLive(true);
    } catch (err: any) {
      setError(`Failed to start broadcast: ${err.message}`);
    }
  };

  const stopBroadcast = async () => {
    if (client) {
      await client.stopBroadcast();
      setIsLive(false);
    }
  };

  return { previewRef, startBroadcast, stopBroadcast, isLive, error };
}
```

Create usePlayer.ts hook following research Pattern 2:
```typescript
import { useState, useEffect, useRef } from 'react';

declare global {
  interface Window {
    IVSPlayer: any;
  }
}

export function usePlayer(sessionId: string, apiBaseUrl: string) {
  const [player, setPlayer] = useState<any>(null);
  const [isPlaying, setIsPlaying] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const videoRef = useRef<HTMLVideoElement>(null);

  useEffect(() => {
    if (!videoRef.current || !window.IVSPlayer) return;

    const initPlayer = async () => {
      try {
        // Fetch playback URL
        const response = await fetch(`${apiBaseUrl}/sessions/${sessionId}/playback`);
        const { playbackUrl, status } = await response.json();

        if (!playbackUrl) {
          setError('No playback URL available');
          return;
        }

        // Initialize player
        const ivsPlayer = window.IVSPlayer.create();
        ivsPlayer.attachHTMLVideoElement(videoRef.current);

        // Event listeners
        ivsPlayer.addEventListener(window.IVSPlayer.PlayerState.PLAYING, () => {
          setIsPlaying(true);
        });
        ivsPlayer.addEventListener(window.IVSPlayer.PlayerState.IDLE, () => {
          setIsPlaying(false);
        });
        ivsPlayer.addEventListener(window.IVSPlayer.PlayerEventType.ERROR, (error: any) => {
          setError(`Player error: ${error}`);
        });

        // Load and play
        ivsPlayer.load(playbackUrl);
        ivsPlayer.play();

        setPlayer(ivsPlayer);
      } catch (err: any) {
        setError(err.message);
      }
    };

    initPlayer();

    return () => {
      if (player) {
        player.pause();
        player.delete();
      }
    };
  }, [sessionId, apiBaseUrl]);

  return { videoRef, player, isPlaying, error };
}
```
