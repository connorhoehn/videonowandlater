From web/src/features/viewer/usePlayer.ts (Phase 3, live playback pattern):
```typescript
import { useEffect, useRef, useState } from 'react';

export function usePlayer(playbackUrl: string) {
  const videoRef = useRef<HTMLVideoElement>(null);
  const playerRef = useRef<any>(null);

  useEffect(() => {
    if (!videoRef.current || !window.IVSPlayer) return;

    const player = window.IVSPlayer.create();
    player.attachHTMLVideoElement(videoRef.current);
    playerRef.current = player;

    player.load(playbackUrl);
    player.setAutoplay(true);

    return () => player.delete();
  }, [playbackUrl]);

  return { videoRef };
}
```

    Add ResponseHeadersPolicy to CloudFront distribution to enable browser HLS manifest fetching.

    **CloudFront CORS Policy (session-stack.ts):**
    In SessionStack constructor, BEFORE CloudFront distribution creation:

    ```typescript
    // CloudFront CORS policy for HLS playback
    const recordingsCorsPolicy = new cloudfront.ResponseHeadersPolicy(this, 'RecordingsCorsPolicy', {
      corsBehavior: {
        accessControlAllowOrigins: { items: ['*'] },
        accessControlAllowMethods: { items: ['GET', 'HEAD', 'OPTIONS'] },
        accessControlAllowHeaders: { items: ['*'] },
        accessControlExposeHeaders: { items: ['*'] },
        originOverride: true,
      },
      comment: 'CORS headers for IVS Player HLS requests',
    });
    ```

    Update CloudFront Distribution defaultBehavior:
    ```typescript
    defaultBehavior: {
      origin: recordingsBucketOrigin,
      viewerProtocolPolicy: cloudfront.ViewerProtocolPolicy.REDIRECT_TO_HTTPS,
      cachePolicy: cloudfront.CachePolicy.CACHING_OPTIMIZED,
      responseHeadersPolicy: recordingsCorsPolicy,  // ADD THIS LINE
    }
    ```

    **Why:** Browser HLS players (IVS Player) make cross-origin requests to fetch .m3u8 manifests and .ts segments from CloudFront. Without CORS headers, browser blocks requests with "CORS policy: No 'Access-Control-Allow-Origin' header" error.

    Import required: `import * as cloudfront from 'aws-cdk-lib/aws-cloudfront';` (already imported in Phase 5)
