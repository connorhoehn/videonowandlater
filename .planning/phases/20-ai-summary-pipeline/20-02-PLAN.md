Create a new React component SummaryDisplay that encapsulates the status-based rendering logic for AI summaries. This component will be reused on recording cards (truncated) and replay info panel (full).

Component signature and implementation:

```typescript
import React from 'react';

export interface SummaryDisplayProps {
  summary?: string;
  status?: 'pending' | 'available' | 'failed';
  truncate?: boolean; // true for cards (2 lines), false for full text
  className?: string; // for styling flexibility
}

export const SummaryDisplay: React.FC<SummaryDisplayProps> = ({
  summary,
  status = 'pending', // Default to pending for backward compatibility
  truncate = false,
  className = '',
}) => {
  // Treat undefined status as 'pending' (pre-Phase 20 sessions)
  const displayStatus = status ?? 'pending';

  if (displayStatus === 'pending') {
    return (
      <p className={`text-gray-500 text-sm ${className}`}>
        Summary coming soon...
      </p>
    );
  }

  if (displayStatus === 'available' && summary) {
    return (
      <p className={`text-sm ${truncate ? 'line-clamp-2' : ''} ${className}`}>
        {summary}
      </p>
    );
  }

  if (displayStatus === 'failed') {
    return (
      <p className={`text-gray-400 text-sm italic ${className}`}>
        Summary unavailable
      </p>
    );
  }

  // Fallback for undefined/unknown status
  return null;
};
```

Key features:
- `status` defaults to 'pending' (backward compatible with undefined)
- `truncate` prop controls line-clamp behavior (2 lines on cards, none on panels)
- `className` prop allows parent components to add additional styling
- Handles all three status states: pending, available, failed
- Gracefully returns null for unknown states
- Uses Tailwind classes (line-clamp-2, text-gray-500, text-gray-400, text-sm, italic)
