2. Add S3 fetch before Bedrock invocation:
```typescript
import { S3Client, GetObjectCommand } from "@aws-sdk/client-s3";
import { transformToString } from "@aws-sdk/util-utf8-node";

const s3Client = new S3Client({ region: process.env.AWS_REGION });

// Fetch transcript from S3
const [bucketName, key] = transcriptS3Uri.split('/', 3).slice(2);  // Parse s3://bucket/key
const s3Response = await s3Client.send(
  new GetObjectCommand({ Bucket: bucketName, Key: key })
);
const transcriptText = await transformToString(s3Response.Body as any);

if (!transcriptText || transcriptText.trim().length === 0) {
  // Log warning but don't fail - set summary status to failed
  logger.warn(`Empty transcript from S3: ${transcriptS3Uri}`);
  await updateSessionRecording(sessionId, {
    SUMMARY: {
      summaryStatus: 'failed',
      summaryText: null,
      summaryTimestamp: new Date().toISOString(),
    }
  });
  return { statusCode: 200 };
}
```

5. Wrap entire operation in try/catch with non-blocking error handling:
```typescript
try {
  // S3 fetch
  // Bedrock invocation
  // Session update
} catch (error) {

  // Non-blocking: set failed status without throwing

      SUMMARY: {
        summaryStatus: 'failed',
        summaryText: null,
        summaryTimestamp: new Date().toISOString(),
```
