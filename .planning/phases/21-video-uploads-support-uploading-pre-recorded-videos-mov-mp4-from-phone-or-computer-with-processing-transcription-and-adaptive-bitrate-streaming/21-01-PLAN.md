```typescript
export async function createSession(tableName: string, input: CreateSessionInput): Promise<Session> {
  const docClient = getDocumentClient();
  const now = new Date().toISOString();
  const sessionId = generateSessionId();

  const session: Session = {
    sessionId,
    userId: input.userId,
    sessionType: input.sessionType,
    status: SessionStatus.CREATING,
    claimedResources: input.claimedResources,
    createdAt: now,
    version: 1,
  };

  await docClient.send(new PutCommand({
    TableName: tableName,
    Item: {
      PK: `SESSION#${sessionId}`,
      SK: 'METADATA',
      ...session,
    },
  }));

  return session;
}
```

# Upload Session Schema

UPLOAD sessions follow the same DynamoDB schema but with additional fields:

```typescript
export interface UploadSession extends Session {
  uploadId: string;                              // S3 multipart upload ID
  uploadStatus: 'pending' | 'processing' | 'converting' | 'available' | 'failed';
  uploadProgress: number;                        // 0-100 percentage
  uploadSourceLocation?: 'phone' | 'computer';   // Optional metadata for analytics
  sourceFileName: string;                        // Original filename
  sourceFileSize: number;                        // In bytes
  sourceCodec?: string;                          // Detected codec (H.264, H.265, etc)
  mediaConvertJobName?: string;                  // vnl-{sessionId}-{epochMs}
  convertStatus?: 'pending' | 'processing' | 'available' | 'failed';
}
```

State machine for UPLOAD sessions:
```
status: 'creating' + uploadStatus: 'pending'
  → [S3 multipart complete]
status: 'creating' + uploadStatus: 'processing'
  → [MediaConvert job submitted]
status: 'creating' + uploadStatus: 'converting' + convertStatus: 'processing'
  → [MediaConvert job complete]
status: 'ended' + uploadStatus: 'available' + convertStatus: 'available'
```

Critical: UPLOAD sessions remain in `status: 'creating'` until `convertStatus: 'available'`, not when S3 upload completes. This prevents the "session status confusion" pitfall where frontend thinks the session is ready before HLS URL is populated.

```typescript
/**
 * Create a new UPLOAD session with upload tracking initialized
 */
export async function createUploadSession(
  tableName: string,
  userId: string,
  sourceFileName: string,
  sourceFileSize: number,
  sourceCodec?: string
): Promise<Session> {
  const docClient = getDocumentClient();
  const now = new Date().toISOString();
  const sessionId = generateSessionId();

  const session: Session = {
    sessionId,
    userId,
    sessionType: SessionType.UPLOAD,
    status: SessionStatus.CREATING,
    claimedResources: { chatRoom: '' }, // Empty; may be populated later if chat is added
    createdAt: now,
    version: 1,
    uploadStatus: 'pending',
    uploadProgress: 0,
    sourceFileName,
    sourceFileSize,
    sourceCodec,
  };

  await docClient.send(new PutCommand({
    TableName: tableName,
    Item: {
      PK: `SESSION#${sessionId}`,
      SK: 'METADATA',
      ...session,
    },
  }));

  return session;
}

/**
 * Update upload progress after S3 multipart completion or during upload
 * Preserves all other fields unchanged
 */
export async function updateUploadProgress(
  tableName: string,
  sessionId: string,
  uploadStatus: string,
  uploadProgress: number
): Promise<void> {
  const docClient = getDocumentClient();

  await docClient.send(new UpdateCommand({
    TableName: tableName,
    Key: {
      PK: `SESSION#${sessionId}`,
      SK: 'METADATA',
    },
    UpdateExpression: 'SET uploadStatus = :status, uploadProgress = :progress',
    ExpressionAttributeValues: {
      ':status': uploadStatus,
      ':progress': uploadProgress,
    },
  }));
}

/**
 * Update convert status after MediaConvert job submission or completion
 * Stores job name and conversion progress
 */
export async function updateConvertStatus(
  tableName: string,
  sessionId: string,
  mediaConvertJobName: string,
  convertStatus: string
): Promise<void> {
  const docClient = getDocumentClient();

  await docClient.send(new UpdateCommand({
    TableName: tableName,
    Key: {
      PK: `SESSION#${sessionId}`,
      SK: 'METADATA',
    },
    UpdateExpression: 'SET mediaConvertJobName = :jobName, convertStatus = :status',
    ExpressionAttributeValues: {
      ':jobName': mediaConvertJobName,
      ':status': convertStatus,
    },
  }));
}
```
